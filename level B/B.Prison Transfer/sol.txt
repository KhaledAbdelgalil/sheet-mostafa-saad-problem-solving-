#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,t,c;
    cin>>n>>t>>c;
    int cnt=0,cnt1=0;
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        if(x<=t)cnt1++;
        else cnt1=0;
        if(cnt1==c)
        {
            cnt1=cnt1-1;
            cnt++;
        }
    }
    cout<<cnt;
    return 0;
}
